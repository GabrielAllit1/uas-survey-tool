param($tool)
if ($tool -eq "uas") {
    conda activate uas_survey_tool_v2
    $env:CUDA_PATH = "C:\Program Files\NVIDIA GPU Computing Toolkit\CUDA\v12.9"
    $env:PATH = "$env:CUDA_PATH\bin;$env:PATH"
    Write-Output "Activated UAS Survey Tool v2.0"
}
elseif ($tool -eq "arbor") {
    conda activate arbor_analyzer
    $env:CUDA_PATH = "C:\Program Files\NVIDIA GPU Computing Toolkit\CUDA\v12.9"
    $env:PATH = "$env:CUDA_PATH\bin;$env:PATH"
    Write-Output "Activated Arbor Analyzer"
}
else {
    Write-Output "Usage: .\switch_tools.ps1 [uas|arbor]"
}
